def build_url(opts)
        uri = URI::HTTP.build(:host => opts[:host],
              :path => opts[:path]                           
              )
        uri.query_values = opts[:query].map{ |k, v| "#{k}=#{CGI.escape(v.to_s)}" }.join('&') if opts[:query].present?
        uri
end
